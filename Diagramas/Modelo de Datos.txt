-- Creación de la tabla Estudio Contable (no depende de otras)
CREATE TABLE estudio_contable (
    id_estudio INTEGER,
    CONSTRAINT estudio_contable_pkey PRIMARY KEY (id_Estudio)
);

-- Creación de la tabla Contadores
CREATE TABLE contador (
    cuil BIGINT,
    id_estudio INTEGER,
    clerk_id TEXT,
    email TEXT,
    nombre TEXT,
    apellido TEXT,
    teléfono TEXT,
    CONSTRAINT contador_pkey PRIMARY KEY (cuil),
    CONSTRAINT id_estudio FOREIGN KEY (id_estudio) REFERENCES estudio_contable(id_estudio)
);

-- Creación de la tabla Clientes
CREATE TABLE cliente (
    cuil BIGINT,
    id_estudio INTEGER,
    contador BIGINT,
    clerk_id TEXT,
    email TEXT,
    nombre TEXT,
    apellido TEXT,
    teléfono TEXT,
    CONSTRAINT cliente_pkey PRIMARY KEY (cuil),
    CONSTRAINT id_estudio FOREIGN KEY (id_estudio) REFERENCES estudio_contable(id_estudio),
    CONSTRAINT contador_cliente FOREIGN KEY (contador) REFERENCES contador(cuil)
);

-- Creación de la tabla Entidad Tributaria
CREATE TABLE entidad_tributaria (
    id_entidad INTEGER,
    nombre TEXT,
    url TEXT,
    CONSTRAINT entidad_tributaria_pkey PRIMARY KEY (id_entidad)
);

-- Creación de la tabla Impuesto
CREATE TABLE impuesto (
    id_impuesto INTEGER,
    formato text,
    CONSTRAINT impuesto_pkey PRIMARY KEY (id_impuesto)
);

-- Creación de la tabla Comprobante
CREATE TABLE Comprobante (
    numero_boleta INTEGER,
    periodo_fiscal DATE,
    importe REAL,
    CONSTRAINT comprobante_pkey PRIMARY KEY (numero_boleta)
);

-- Creación de la tabla Liquidacion
CREATE TABLE liquidacion (
    numero_boleta INTEGER,
    importe REAL,
    periodo_fiscal DATE,
    estado TEXT,
    id_impuesto INTEGER,
    cuil_cliente INTEGER,
    numero_boleta_comprobante INTEGER UNIQUE,  -- Relación 1:1 (cada liquidación tiene su propio comprobante)
    CONSTRAINT liquidacion_pkey PRIMARY KEY (numero_boleta),
    CONSTRAINT id_impuesto FOREIGN KEY (id_impuesto) REFERENCES impuesto(id_impuesto),
    CONSTRAINT numero_boleta FOREIGN KEY (numero_boleta_comprobante) REFERENCES comprobante(numero_boleta),
    CONSTRAINT cuil_cliente FOREIGN KEY (cuil_cliente) REFERENCES cliente(cuil)
);

-- Tabla Turno (Relación entre Clientes y Contadores con Fecha/Hora)
CREATE TABLE turno (
    cuil_cliente INTEGER,
    cuil_contador INTEGER,
    fecha DATE,
    hora TIME,
    CONSTRAINT pk_turno PRIMARY KEY (cuil_cliente, cuil_contador, fecha, hora),
    CONSTRAINT cuil_cliente FOREIGN KEY (cuil_cliente) REFERENCES cliente(cuil),
    CONSTRAINT cuil_contador FOREIGN KEY (cuil_contador) REFERENCES contador(cuil)
);

-- Tabla Inscripto_en (Relación entre Clientes y Entidades Tributarias)
CREATE TABLE inscripto_en (
    cuil_cliente INTEGER,
    id_entidad INTEGER,
    clave TEXT,
    CONSTRAINT inscripto_en_pkey PRIMARY KEY (cuil_cliente, id_entidad),
    CONSTRAINT cuil_cliente FOREIGN KEY (cuil_cliente) REFERENCES cliente(cuil),
    CONSTRAINT id_entidad FOREIGN KEY (id_entidad) REFERENCES entidad_tributaria(id_entidad)
);

-- Nota sobre autenticación:
-- La autenticación se maneja externamente con Clerk.
-- Los usuarios se registran e identifican mediante Clerk, y se vincula el clerk_id
-- para asociar la identidad de Clerk con los Clientes o Contadores
-- datos del sistema. No se almacenan contraseñas en la base de datos del sistema.
